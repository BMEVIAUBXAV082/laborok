# Boto3 erőforrások törlése

```python
import boto3, time

# ── Azonosítók (ugyanazok, mint F1–F7-ben) ──────────────────────────────────
NEPTUN   = "______"   # TODO: Neptun-kódod
REGION   = "us-east-1"
PREFIX   = f"bme-de-{NEPTUN.lower()}"

BUCKET_NAMES = [
    f"{PREFIX}-smart-city-raw",
    f"{PREFIX}-smart-city-processed",
    f"{PREFIX}-smart-city-streaming",
    f"{PREFIX}-smart-city-query-results",
]
DB_NAME      = f"smart_city_{NEPTUN.lower()}"
CRAWLER_NAME = f"{PREFIX}-sensor-crawler"
JOB_NAME     = f"{PREFIX}-sensor-bronze-to-silver"
WORKGROUP    = f"{PREFIX}-smart-city-wg"
STREAM_NAME  = f"{PREFIX}-smart-city-events"
FN_NAME      = f"{PREFIX}-iot-processor"
FH_NAME      = f"{PREFIX}-iot-firehose"

s3       = boto3.client("s3",       region_name=REGION)
glue     = boto3.client("glue",     region_name=REGION)
athena   = boto3.client("athena",   region_name=REGION)
kinesis  = boto3.client("kinesis",  region_name=REGION)
lmb      = boto3.client("lambda",   region_name=REGION)
firehose = boto3.client("firehose", region_name=REGION)


def try_delete(label, fn, **kwargs):
    try:
        fn(**kwargs)
        print(f"  ✓ Törölve: {label}")
    except Exception as e:
        print(f"  – Kihagyva ({label}): {e}")


# ── 1. Firehose ──────────────────────────────────────────────────────────────
print("1. Firehose törlése...")
try_delete(FH_NAME, firehose.delete_delivery_stream, DeliveryStreamName=FH_NAME)
time.sleep(3)

# ── 2. Lambda event source mapping + function ────────────────────────────────
print("2. Lambda törlése...")
try:
    mappings = lmb.list_event_source_mappings(FunctionName=FN_NAME)["EventSourceMappings"]
    for m in mappings:
        try_delete(f"mapping {m['UUID']}", lmb.delete_event_source_mapping, UUID=m["UUID"])
except Exception as e:
    print(f"  – Event source mappings: {e}")
try_delete(FN_NAME, lmb.delete_function, FunctionName=FN_NAME)

# ── 3. Kinesis stream ────────────────────────────────────────────────────────
print("3. Kinesis stream törlése...")
try_delete(STREAM_NAME, kinesis.delete_stream, StreamName=STREAM_NAME, EnforceConsumerDeletion=True)

# ── 4. Athena WorkGroup ──────────────────────────────────────────────────────
print("4. Athena WorkGroup törlése...")
try_delete(WORKGROUP, athena.delete_work_group, WorkGroup=WORKGROUP, RecursiveDeleteOption=True)

# ── 5. Glue: job + crawler + táblák + database ───────────────────────────────
print("5. Glue erőforrások törlése...")
try_delete(JOB_NAME,     glue.delete_job,     JobName=JOB_NAME)
try_delete(CRAWLER_NAME, glue.delete_crawler, Name=CRAWLER_NAME)
try:
    tables = glue.get_tables(DatabaseName=DB_NAME)["TableList"]
    for t in tables:
        try_delete(f"tábla {t['Name']}", glue.delete_table,
                   DatabaseName=DB_NAME, Name=t["Name"])
except Exception as e:
    print(f"  – Táblák listázása: {e}")
try_delete(DB_NAME, glue.delete_database, Name=DB_NAME)

# ── 6. S3 bucket-ek kiürítése és törlése ────────────────────────────────────
print("6. S3 bucket-ek törlése...")
for bucket in BUCKET_NAMES:
    try:
        paginator = s3.get_paginator("list_objects_v2")
        for page in paginator.paginate(Bucket=bucket):
            objects = page.get("Contents", [])
            if objects:
                s3.delete_objects(
                    Bucket=bucket,
                    Delete={"Objects": [{"Key": o["Key"]} for o in objects]},
                )
        s3.delete_bucket(Bucket=bucket)
        print(f"  ✓ Törölve: {bucket}")
    except Exception as e:
        print(f"  – Kihagyva ({bucket}): {e}")

print("\n✓ Boto3 cleanup kész.")
```
