### 3.1 Glue kliens + változók

```python
import boto3, time

REGION   = 'us-east-1'
NEPTUN   = "______"   # TODO: Neptun-kódod
PREFIX   = f"bme-de-{NEPTUN.lower()}"
BUCKETS  = {
    "raw":       f"{PREFIX}-smart-city-raw",
    "processed": f"{PREFIX}-smart-city-processed",
}

sts        = boto3.client('sts', region_name=REGION)
ACCOUNT_ID = sts.get_caller_identity()["Account"]
LAB_ROLE   = f"arn:aws:iam::{ACCOUNT_ID}:role/LabRole"

glue = boto3.client('glue', region_name=REGION)
DB_NAME      = f"smart_city_{NEPTUN.lower()}"
CRAWLER_NAME = f"{PREFIX}-sensor-crawler"
```

### 3.2 Glue adatbázis létrehozása

```python
try:
    glue.create_database(
        DatabaseInput={
            "Name":        DB_NAME,
            "Description": "Smart City IoT szenzor adatok katalógusa",
        }
    )
    print(f"✓ Adatbázis létrehozva: {DB_NAME}")
except glue.exceptions.AlreadyExistsException:
    print(f"  Már létezik: {DB_NAME}")
```

### 3.3 Glue Crawler létrehozása

```python
try:
    glue.create_crawler(
        Name=CRAWLER_NAME,
        Role=None,           # TODO: LAB_ROLE
        DatabaseName=None,   # TODO: DB_NAME
        Description="Smart City szenzor CSV-k felderítése",
        Targets={
            "S3Targets": [{
                "Path": f"s3://{BUCKETS['raw']}/sensors/",
            }]
        },
        SchemaChangePolicy={
            "UpdateBehavior": "UPDATE_IN_DATABASE",
            "DeleteBehavior": "LOG",
        },
    )
    print(f"✓ Crawler létrehozva: {CRAWLER_NAME}")
except glue.exceptions.AlreadyExistsException:
    print(f"  Crawler már létezik: {CRAWLER_NAME}")
```

### 3.4 Crawler futtatása + polling

```python
glue.start_crawler(Name=CRAWLER_NAME)
print(f"Crawler elindítva: {CRAWLER_NAME}")
print("Várakozás... (kb. 1-2 perc)")

while True:
    info  = glue.get_crawler(Name=CRAWLER_NAME)["Crawler"]
    state = info["State"]
    print(f"  Státusz: {state}")
    if state == "READY":   # a crawler végzett
        break
    time.sleep(15)

last_crawl = info.get("LastCrawl", {})
print(f"\n✓ Crawler kész!")
print(f"  Összesítő  : {last_crawl.get('Summary', 'n/a')}")
print(f"  Tábla szám : {last_crawl.get('TablesCreated', 0)} létrehozva, {last_crawl.get('TablesUpdated', 0)} frissítve")
```

### 3.5 Felderített séma megjelenítése

```python
import pandas as pd

tables = glue.get_tables(DatabaseName=DB_NAME)["TableList"]
print(f"Felderített táblák ({len(tables)} db):\n")

for table in tables:
    cols    = table["StorageDescriptor"]["Columns"]
    partitions = [p["Name"] for p in table.get("PartitionKeys", [])]
    print(f"Tábla: {table['Name']}")
    print(f"  S3 helye  : {table['StorageDescriptor']['Location']}")
    print(f"  Partíciók : {partitions}")
    print(f"  Oszlopok  :")
    for col in cols:
        print(f"    {col['Name']:20s} {col['Type']}")
    print()

assert len(tables) > 0, "HIBA: A crawler nem talált táblákat!"
```
