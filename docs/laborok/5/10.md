# Feladat 9 – Terraform modulok: Kinesis + Lambda

### 9.1 Streaming modul: `modules/streaming/variables.tf`

```hcl
variable "project_name" { type = string }
variable "neptun_code"  { type = string }
variable "environment"  { type = string }
variable "lab_role_arn" { type = string }

variable "shard_count" {
  type    = number
  default = 1
}

variable "target_bucket_id" {
  type        = string
  description = "streaming S3 bucket neve"
}

variable "target_bucket_arn" {
  type        = string
  description = "streaming S3 bucket ARN-ja"
}

variable "buffer_seconds" {
  type    = number
  default = 60
}
```

### 9.2 Streaming modul: `modules/streaming/main.tf`

```hcl
locals {
  prefix = "${var.project_name}-${var.neptun_code}"
}

# ── Kinesis Data Streams ─────────────────────────────────────────────────────
resource "aws_kinesis_stream" "iot_events" {
  name        = "${local.prefix}-smart-city-events-tf"
  shard_count = var.shard_count

  stream_mode_details {
    stream_mode = "PROVISIONED"
  }
}

# ── Lambda function (archive_file data source) ───────────────────────────────
data "archive_file" "lambda_zip" {
  type        = "zip"
  source_file = "${path.module}/lambda_processor.py"
  output_path = "${path.module}/lambda_processor.zip"
}

resource "aws_lambda_function" "iot_processor" {
  function_name    = "${local.prefix}-iot-processor-tf"
  role             = var.lab_role_arn
  handler          = "lambda_processor.lambda_handler"
  runtime          = "python3.12"
  timeout          = 60
  memory_size      = 128
  filename         = data.archive_file.lambda_zip.output_path
  source_code_hash = data.archive_file.lambda_zip.output_base64sha256
  description      = "Smart City IoT esemény validátor (Terraform IaC)"
}

resource "aws_lambda_event_source_mapping" "kinesis_trigger" {
  event_source_arn  = aws_kinesis_stream.iot_events.arn
  function_name     = aws_lambda_function.iot_processor.arn
  starting_position = "LATEST"
  batch_size        = 25
}

# ── Kinesis Firehose ─────────────────────────────────────────────────────────
resource "aws_kinesis_firehose_delivery_stream" "iot_to_s3" {
  name        = "${local.prefix}-iot-firehose-tf"
  destination = "extended_s3"

  extended_s3_configuration {
    role_arn           = var.lab_role_arn
    bucket_arn         = var.target_bucket_arn
    prefix             = "sensors/year=!{timestamp:yyyy}/month=!{timestamp:MM}/day=!{timestamp:dd}/"
    error_output_prefix = "errors/"

    buffering_size     = 5
    buffering_interval = var.buffer_seconds
    compression_format = "GZIP"
  }
}
```

### 9.3 Streaming modul: `modules/streaming/outputs.tf`


```hcl
output "stream_name" {
  value = aws_kinesis_stream.iot_events.name
}

output "stream_arn" {
  value = aws_kinesis_stream.iot_events.arn
}

output "lambda_arn" {
  value = aws_lambda_function.iot_processor.arn
}

output "firehose_name" {
  value = aws_kinesis_firehose_delivery_stream.iot_to_s3.name
}
```

!!! warning "lambda_processor.py"
    Másold a `week07/labor/scripts/lambda_processor.py` fájlt a `week07/labor/terraform/modules/streaming/` mappába is, mert az `archive_file` data source onnan olvassa.

```bash
cp week07/labor/scripts/lambda_processor.py week07/labor/terraform/modules/streaming/
```

### 9.4 Module hozzáadása a root `main.tf`-hez

```hcl
module "streaming" {
  source      = "./modules/streaming"

  project_name     = var.project_name
  neptun_code      = var.neptun_code
  environment      = local.environment
  lab_role_arn     = local.lab_role_arn
  shard_count      = var.shard_count
  target_bucket_id  = aws_s3_bucket.smart_city["streaming"].id
  target_bucket_arn = aws_s3_bucket.smart_city["streaming"].arn
  buffer_seconds   = var.firehose_buffer_seconds
}
```

### 9.5 Staging workspace

```bash
cd week07/labor/docker

# Új workspace létrehozása
docker compose run --rm terraform workspace new staging
docker compose run --rm terraform workspace list

# Apply staging-ban (terraform.tfvars automatikusan betöltődik; environment = terraform.workspace = "staging")
docker compose run --rm terraform apply

# State lista megjelenítése
docker compose run --rm terraform state list
```
