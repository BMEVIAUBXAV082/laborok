### 5.2 Segédfüggvény: lekérdezés futtatása

!!! warning "Jupyter TODO-k és az `___` változó"
    IPython/Jupyter-ben a `_`, `__`, `___` speciális változók (előző cellák kimenete). Soha ne hagyd kitöltetlenül a `___` helyőrzőket — nem hibát adnak, hanem váratlan értéket kapnak! A segédfüggvényt add hozzá a notebookhoz pontosan így:

```python
def run_athena_query(sql: str, description: str = "") -> pd.DataFrame:
    """Athena SQL futtatása → pandas DataFrame."""
    if description:
        print(f"\n--- {description} ---")

    resp     = athena.start_query_execution(
        QueryString=sql,
        WorkGroup=WORKGROUP,
        QueryExecutionContext={"Database": DB_NAME},
    )
    query_id = resp["QueryExecutionId"]
    print(f"  Query ID: {query_id}")

    # Polling: SUCCEEDED / FAILED / CANCELLED
    while True:
        info   = athena.get_query_execution(QueryExecutionId=query_id)
        status = info["QueryExecution"]["Status"]["State"]
        if status in ("SUCCEEDED", "FAILED", "CANCELLED"):
            break
        time.sleep(2)

    if status != "SUCCEEDED":
        reason = info["QueryExecution"]["Status"].get("StateChangeReason", "ismeretlen")
        raise RuntimeError(f"Athena hiba ({status}): {reason}")

    result = athena.get_query_results(QueryExecutionId=query_id)
    rows   = result["ResultSet"]["Rows"]
    header = [c["VarCharValue"] for c in rows[0]["Data"]]
    data   = [
        [cell.get("VarCharValue", "") for cell in row["Data"]]
        for row in rows[1:]
    ]
    df = pd.DataFrame(data, columns=header)
    print(f"  Eredmény: {len(df)} sor")
    return df
```

### 5.3 Lekérdezés 1 – Legrosszabb légminőség napok

!!! note "Timestamp kezelés Athena-ban"
    A Glue Crawler a `timestamp` oszlopot `string` típusként látja a CSV-ben. Az Athena `DATE()` és `HOUR()` függvények nem tudnak közvetlenül ISO 8601 stringet kezelni — ezért `from_iso8601_timestamp()` konverziót kell alkalmazni.

```python
sql_pm25 = """
SELECT
    location,
    DATE(from_iso8601_timestamp(timestamp))             AS nap,
    ROUND(AVG(metric_value), 2)                         AS avg_pm25_ugm3,
    ROUND(MAX(metric_value), 2)                         AS max_pm25_ugm3
FROM sensors
WHERE metric_name = 'PM2.5'
GROUP BY location, DATE(from_iso8601_timestamp(timestamp))
ORDER BY avg_pm25_ugm3 DESC
LIMIT 10
"""

df_pm25 = run_athena_query(sql_pm25, "Top 10 legszennyezettebb nap (PM2.5)")
display(df_pm25)
```

sql_noise = """
SELECT
    hour(from_iso8601_timestamp(timestamp))              AS ora,
    location,
    ROUND(AVG(metric_value), 1)                          AS avg_zajszint_db,

### 5.5 Lekérdezés 3 – Forgalmi torlódások

```python
sql_traffic = """
WITH agg AS (
    SELECT
        location,
        DATE(from_iso8601_timestamp(timestamp))                                    AS nap,
        hour(from_iso8601_timestamp(timestamp))                                    AS ora,
        ROUND(AVG(CASE WHEN metric_name='vehicle_count' THEN metric_value END), 0) AS avg_jarmu,
        ROUND(AVG(CASE WHEN metric_name='avg_speed'     THEN metric_value END), 1) AS avg_sebesseg_kmh
    FROM sensors
    WHERE metric_name IN ('vehicle_count', 'avg_speed')
      AND device_type IN ('traffic', 'combined')
    GROUP BY location,
             DATE(from_iso8601_timestamp(timestamp)),
             hour(from_iso8601_timestamp(timestamp))
)
SELECT * FROM agg
WHERE avg_sebesseg_kmh < 25
  AND avg_jarmu > 150
ORDER BY avg_sebesseg_kmh ASC
LIMIT 20
"""

df_traffic = run_athena_query(sql_traffic, "Forgalmi torlódások (sebesség<25 km/h, jármű>150/h)")
display(df_traffic)
```
