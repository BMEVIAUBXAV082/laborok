# 6. feladat: Adatminőség validáció

A valódi adatpipeline-okban az adatminőség folyamatos ellenőrzése kritikus. Hibás, hiányzó vagy inkonzisztens adatok néma hibákat okozhatnak a downstream analitikában. Ebben a feladatban szisztematikus adatminőség-ellenőrző lekérdezéseket írunk DuckDB-vel: null értékeket keresünk, duplikátumokat azonosítunk, referenciális integritást ellenőrzünk, és értéktartomány-validálást végzünk.

### Notebook létrehozása

1. Hozzunk létre egy új notebookot `feladat6_dataquality.ipynb` néven.

2. Az első cellában töltsük be a staging táblákat:

    ```python

    import pandas as pd

    duck = duckdb.connect()
    RAW = "/home/jovyan/data/extract/full"

    tables = ["customers", "products", "orders", "order_items"]
    for table in tables:
        duck.execute(f"""
            CREATE OR REPLACE TABLE stg_{table}
            AS SELECT * FROM read_parquet('{RAW}/{table}.parquet')
        """)
    print("Staging táblák betöltve.")
    ```

### Null értékek vizsgálata

3. Ellenőrizzük az összes táblában az null értékeket oszloponként:

    ```python
    print("=== Null értékek vizsgálata ===\n")
    null_results = []

    checks = {
        "stg_customers":   ["customer_id", "name", "email", "city", "segment", "created_at"],
        "stg_products":    ["product_id", "name", "category", "price", "is_active"],
        "stg_orders":      ["order_id", "customer_id", "order_date", "status", "total_amount"],
        "stg_order_items": ["order_id", "product_id", "quantity", "unit_price"],
    }

    for table, cols in checks.items():
        for col in cols:
            null_cnt = duck.execute(
                f"SELECT COUNT(*) - COUNT({col}) FROM {table}"
            ).fetchone()[0]
            null_results.append({
                "tábla": table,
                "oszlop": col,
                "null_count": null_cnt,
                "státusz": "✓" if null_cnt == 0 else "✗ HIBA",
            })

    null_df = pd.DataFrame(null_results)
    errors = null_df[null_df["null_count"] > 0]
    if errors.empty:
        print("Minden kötelező mező kitöltött – nincs null érték. ✓")
    else:
        print(errors.to_string(index=False))
    print(f"\nEllenőrzött cellák száma: {len(null_df)}")
    ```

### Duplikált sorok keresése

4. Ellenőrizzük, hogy az elsődleges kulcsok valóban egyediek-e:

    ```python
    print("=== Duplikátum ellenőrzés (PK egyediség) ===\n")
    pk_checks = [
        ("stg_customers",   "customer_id"),
        ("stg_products",    "product_id"),
        ("stg_orders",      "order_id"),
    ]

    dup_results = []
    for table, pk in pk_checks:
        dup_cnt = duck.execute(f"""
            SELECT COUNT(*) FROM (
                SELECT {pk}, COUNT(*) AS cnt
                FROM {table}
                GROUP BY {pk}
                HAVING COUNT(*) > 1
            )
        """).fetchone()[0]
        dup_results.append({
            "tábla": table,
            "pk_oszlop": pk,
            "duplikált_kulcsok": dup_cnt,
            "státusz": "✓" if dup_cnt == 0 else "✗ HIBA",
        })

    print(pd.DataFrame(dup_results).to_string(index=False))
    ```

### Referenciális integritás ellenőrzése

5. Ellenőrizzük, hogy minden FK-hivatkozás érvényes rekordra mutat-e:

    ```python
    print("=== Referenciális integritás (FK ellenőrzés) ===\n")
    fk_checks = [
        ("stg_orders",      "customer_id", "stg_customers", "customer_id",
         "Rendelés → Ügyfél"),
        ("stg_order_items", "order_id",    "stg_orders",    "order_id",
         "Rendelési tétel → Rendelés"),
        ("stg_order_items", "product_id",  "stg_products",  "product_id",
         "Rendelési tétel → Termék"),
    ]

    fk_results = []
    for child_tbl, child_col, parent_tbl, parent_col, desc in fk_checks:
        orphan_cnt = duck.execute(f"""
            SELECT COUNT(*) FROM {child_tbl} c
            LEFT JOIN {parent_tbl} p ON c.{child_col} = p.{parent_col}
            WHERE p.{parent_col} IS NULL
        """).fetchone()[0]
        fk_results.append({
            "kapcsolat": desc,
            "árva_sorok": orphan_cnt,
            "státusz": "✓" if orphan_cnt == 0 else "✗ HIBA",
        })

    print(pd.DataFrame(fk_results).to_string(index=False))
    ```

### Értéktartomány ellenőrzés

6. Ellenőrizzük az üzleti szempontból érvényes értéktartományokat:

    ```python
    print("=== Értéktartomány ellenőrzések ===\n")
    range_results = []

    range_checks = [
        ("stg_products",    "price <= 0",        "Érvénytelen ár (price <= 0)"),
        ("stg_order_items", "quantity <= 0",     "Érvénytelen mennyiség"),
        ("stg_order_items", "unit_price <= 0",   "Érvénytelen egységár"),
        ("stg_orders",      "total_amount <= 0", "Érvénytelen rendelési összeg"),
        ("stg_customers",   "segment NOT IN ('standard', 'gold', 'premium')",
         "Érvénytelen szegmens érték"),
    ]

    for table, condition, description in range_checks:
        cnt = duck.execute(
            f"SELECT COUNT(*) FROM {table} WHERE {condition}"
        ).fetchone()[0]
        range_results.append({
            "ellenőrzés": description,
            "hibás_sorok": cnt,
            "státusz": "✓" if cnt == 0 else "✗ HIBA",
        })

    range_df = pd.DataFrame(range_results)
    print(range_df.to_string(index=False))

    # Összesítő
    total_errors = range_df["hibás_sorok"].sum()
    print(f"\nÖsszes értéktartomány-hiba: {total_errors}")
    ```
