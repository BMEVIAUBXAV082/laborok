    1. feladat – Apache Airflow¶

    Az Apache Airflow 2014-ben született az Airbnb-nél, ma a legelterjedtebb nyílt forráskódú workflow-orkesztrátor. A pipeline-ok DAG-ként (Directed Acyclic Graph) definiálhatók Python fájlokban: a csúcspontok a taskok, az élek a végrehajtási sorrendet definiálják.

    Airflow alapfogalmak

    • DAG – pipeline gráfos leírása Python fájlban. Tartalmazza az ütemezést és a task definíciókat.
    • Task / Operator – pipeline egy elemi lépése. PythonOperator, BashOperator, BranchPythonOperator, stb.
    • DAG Run – egy DAG egyetlen lefutása adott execution_date-tel.
    • XCom – „cross-communication": kis méretű értékek átadása taskok között a metaadat-adatbázison keresztül (max ~48 KB). Nagy adatnál csak a fájlútvonalat tároljuk XCom-ban.
    • LocalExecutor – taskok a scheduler folyamatán belül, külön szálakban futnak. Kis–közepes terhelésre elegendő.
    • TaskGroup – taskok vizuális csoportosítása az Airflow UI Graph nézetében.
    • BranchPythonOperator – feltételes elágazás: a callable visszaadja a következő futtatandó task_id-t.
    • catchup=False – nem futtatja vissza a múltbeli hiányzó napokat.

    Az első DAG – hello_world¶

    1. A dags/ mappában hozzuk létre a hello_world.py fájlt (szövegszerkesztővel vagy VS Code-dal).

    2. A fájl tartalma:

from airflow.operators.python import PythonOperator
from airflow.operators.bash import BashOperator
from datetime import datetime, timedelta

default_args = {
    'owner': 'labor05',
    'retries': 1,
    'retry_delay': timedelta(minutes=2),
}

with DAG(
    dag_id='hello_world',
    default_args=default_args,
    description='Első DAG – hello world',
    schedule='@daily',
    start_date=datetime(2024, 1, 1),
    catchup=False,
    tags=['labor05', 'demo'],
) as dag:

    def greet(**context):
        # context['ds']: az execution_date YYYY-MM-DD formátumban
        # Ez NEM az aktuális időpont, hanem a logikai futtatási dátum!
        execution_date = context['ds']
        print(f"Hello, Airflow! Futás dátuma: {execution_date}")
        print(f"DAG ID: {context['dag'].dag_id}")
        return "koszontes_kesz"   # return érték automatikusan XCom-ba kerül

    task_koszontes = PythonOperator(
        task_id='koszontes',
        python_callable=greet,   # Csak referencia – az Airflow hívja meg!
    )

    task_datum = BashOperator(
        task_id='datum_kiiras',
        bash_command='echo "Szerver idő: $(date +%Y-%m-%d\\\\ %H:%M:%S)"',
    )

    task_koszontes >> task_datum   # >> operátor: bal oldali task fut először

      • Graph fülön: koszontes → datum_kiiras. Zöld = sikeres.

    TaskGroup – taskok vizuális csoportosítása¶

    1. Az etl_orders DAG kiegészíthető TaskGroup-okkal, amelyek az Airflow UI-ban összecsukható csoportokat alkotnak. Hozzuk létre a dags/etl_orders_grouped.py fájlt:

      from airflow import DAG
from airflow.operators.python import PythonOperator
from airflow.utils.task_group import TaskGroup
from datetime import datetime, timedelta
import pandas as pd
from sqlalchemy import create_engine
import os

OLTP_CONN = "postgresql://labor:labor@postgres-oltp:5432/oltp_source"
DATA_DIR  = "/opt/airflow/data"

# Az etl_orders-ből ismert függvények (rövidítve)
def extract(**ctx):
    os.makedirs(DATA_DIR, exist_ok=True)
    engine = create_engine(OLTP_CONN)
    df = pd.read_sql(
        "SELECT o.id AS order_id, o.customer_id, o.order_date, o.status, "
        "SUM(oi.quantity * oi.unit_price) AS total_amount "
        "FROM orders o JOIN order_items oi ON oi.order_id = o.id "
        "WHERE o.status = 'completed' GROUP BY o.id, o.customer_id, o.order_date, o.status",
        engine
    )
    path = f"{DATA_DIR}/grp_extract_{ctx['ds_nodash']}.parquet"
    df.to_parquet(path, index=False)
    print(f"Extraktálva: {len(df)} sor")
    return path

def validate(**ctx):
    """Egyszerű DQ ellenőrzés: NULL és negatív értékek."""
    path = ctx['ti'].xcom_pull(task_ids='ingestion.extract')
    df = pd.read_parquet(path)
    null_cnt = df['customer_id'].isna().sum()
    neg_cnt  = (df['total_amount'] < 0).sum()
    print(f"NULL customer_id: {null_cnt} | Negatív összeg: {neg_cnt}")
    if null_cnt > 0 or neg_cnt > 0:
        raise ValueError(f"DQ ellenőrzés SIKERTELEN: {null_cnt} NULL, {neg_cnt} negatív")
    print("DQ ellenőrzés OK")
    return path

def aggregate(**ctx):
    path = ctx['ti'].xcom_pull(task_ids='ingestion.validate')
    df = pd.read_parquet(path)
    df['order_date'] = pd.to_datetime(df['order_date'])
    df['year_month'] = df['order_date'].dt.to_period('M').astype(str)
    agg = df.groupby('year_month').agg(
        order_count=('order_id', 'count'), revenue=('total_amount', 'sum')
    ).reset_index()
    out = path.replace('grp_extract_', 'grp_agg_')
    agg.to_parquet(out, index=False)
    return out

def publish(**ctx):
    path = ctx['ti'].xcom_pull(task_ids='transformation.aggregate')
    df = pd.read_parquet(path)
    print(df.to_string(index=False))

with DAG(
    dag_id='etl_orders_grouped',
    schedule='@daily',
    start_date=datetime(2024, 1, 1),
    catchup=False,
    tags=['labor05', 'etl', 'taskgroup'],
) as dag:

    # TaskGroup 1: adatbefogadás
    with TaskGroup(group_id='ingestion') as tg_ingestion:
        t_extract  = PythonOperator(task_id='extract',  python_callable=extract)
        t_validate = PythonOperator(task_id='validate', python_callable=validate)
        t_extract >> t_validate

    # TaskGroup 2: transzformáció
    with TaskGroup(group_id='transformation') as tg_transform:
        t_agg = PythonOperator(task_id='aggregate', python_callable=aggregate)

    # TaskGroup 3: publikálás
    with TaskGroup(group_id='serving') as tg_serving:
        t_pub = PythonOperator(task_id='publish', python_callable=publish)

tg_ingestion >> tg_transform >> tg_serving

      TaskGroup és XCom – task_ids formátum

      TaskGroup-on belüli task XCom-jára hivatkozáskor a task_id teljes neve: 'csoport_neve.task_neve' – például 'ingestion.extract'. Ha ezt elfelejtjük, az xcom_pull() None-t ad vissza és a Parquet olvasás FileNotFoundError-t dob.

    2. Az Airflow UI-ban aktiváljuk az etl_orders_grouped DAG-ot. A Graph nézetben az összecsukható TaskGroup dobozok (ingestion, transformation, serving) láthatók.

    Jinja templating és ütemezés¶

    1. Az Airflow DAG-ok Jinja template-eket is támogatnak. Hozzuk létre a dags/templated_example.py fájlt:

      from airflow import DAG
from airflow.operators.bash import BashOperator
from airflow.operators.python import PythonOperator
from datetime import datetime

with DAG(
    dag_id='templated_example',
    schedule='0 6 * * 1',    # Minden hétfőn 06:00-kor
    start_date=datetime(2024, 1, 1),
    catchup=False,
    tags=['labor05', 'demo'],
) as dag:

    # {{ ds }}: execution_date YYYY-MM-DD formátumban – Jinja template.
    # Az Airflow a parancs futtatása ELŐTT helyettesíti be az értéket.
    task_heti = BashOperator(
        task_id='heti_riport',
        bash_command='echo "Heti riport: {{ ds }} – {{ dag.dag_id }}"',
    )

    def weekly_summary(**context):
        logical_date = context['logical_date']
        week_number  = logical_date.isocalendar()[1]   # ISO hétszám
        print(f"{logical_date.year}. {week_number}. hét összesítője")
        print(f"Execution date: {context['ds']}")

    task_python = PythonOperator(
        task_id='python_summary',
        python_callable=weekly_summary,
    )

    task_heti >> task_python
