### 6.1 Kinesis stream létrehozása

```python
import boto3, time, json, random

REGION      = 'us-east-1'
NEPTUN      = "______"   # TODO
PREFIX      = f"bme-de-{NEPTUN.lower()}"
STREAM_NAME = f"{PREFIX}-smart-city-events"

kinesis = boto3.client('kinesis', region_name=REGION)

try:
    kinesis.create_stream(
        StreamName=STREAM_NAME,   # a fentebb definiált változó
        ShardCount=1,             # TODO: módosítsd, ha több shard kell (max 10 Learner Lab-ban)
    )
    print(f"✓ Stream létrehozva: {STREAM_NAME}")
except kinesis.exceptions.ResourceInUseException:
    print(f"  Stream már létezik: {STREAM_NAME}")

# ACTIVE állapot megvárása
print("Várakozás ACTIVE állapotra...")
for _ in range(30):
    info        = kinesis.describe_stream_summary(StreamName=STREAM_NAME)
    summary     = info["StreamDescriptionSummary"]
    status      = summary["StreamStatus"]
    shard_count = summary.get("OpenShardCount", "?")   # CREATING közben hiányozhat
    print(f"  Státusz: {status}  ({shard_count} shard)")
    if status == "ACTIVE":
        break
    time.sleep(3)

assert status == "ACTIVE", f"HIBA: Stream nem lett ACTIVE, hanem: {status}"
print(f"\n✓ Stream aktív: {STREAM_NAME}")
```

### 6.3 Batch küldés: 100 szenzor esemény

```python
LOCATIONS    = ["Belváros", "Józsefváros", "Ferencváros", "Kőbánya", "Újpest"]
SENSOR_TYPES = ["air_quality", "noise", "traffic"]
METRICS = {
    "air_quality": [("PM2.5", "ug/m3", 5.0, 150.0)],
    "noise":       [("noise_db", "dB", 35.0, 95.0)],
    "traffic":     [("vehicle_count", "veh/h", 10, 800)],
}

random.seed(None)  # Valódi véletlen a streaming részhez

records = []
for _ in range(100):
    stype   = random.choice(SENSOR_TYPES)
    metric  = random.choice(METRICS[stype])
    mname, unit, lo, hi = metric
    evt = {
        "event_id":    str(uuid.uuid4()),
        "device_id":   f"{stype[:2].upper()}-{random.randint(1,5):03d}",
        "device_type": stype,
        "location":    random.choice(LOCATIONS),
        "metric_name": mname,
        "metric_value": round(random.uniform(lo, hi), 2),
        "unit":        unit,
        "quality_flag": random.choices(["GOOD","SUSPECT","BAD"], weights=[85,12,3])[0],
        "timestamp":   datetime.now(timezone.utc).isoformat(),
    }
    records.append({
        "Data":         json.dumps(evt).encode("utf-8"),
        "PartitionKey": evt["device_id"],   # Shard-elosztás kulcsa
    })

response = kinesis.put_records(
    StreamName=None,   # TODO: STREAM_NAME
    Records=None,      # TODO: records
)

failed = response["FailedRecordCount"]
print(f"Küldés eredménye: {100 - failed}/100 sikeres, {failed} sikertelen")
assert failed == 0, f"HIBA: {failed} rekord nem sikerült!"
print("✓ Mind a 100 rekord sikeresen elküldve!")
```

### 6.4 Rekordok visszaolvasása ellenőrzésképpen

```python
# Shard iterator lekérése (TRIM_HORIZON = legelső rekordtól)
iterator_resp = kinesis.get_shard_iterator(
    StreamName=STREAM_NAME,
    ShardId="shardId-000000000000",
    ShardIteratorType="TRIM_HORIZON",
)
iterator = iterator_resp["ShardIterator"]

# Rekordok kiolvasása
result  = kinesis.get_records(ShardIterator=iterator, Limit=5)
records_read = result["Records"]

print(f"Visszaolvasott rekordok (első 5):")
for r in records_read:
    data = json.loads(r["Data"])
    print(f"  [{data['device_id']}] {data['metric_name']}: {data['metric_value']} {data['unit']}  @ {data['location']}")

assert len(records_read) > 0, "HIBA: Nem sikerült rekordokat visszaolvasni!"
print("\n✓ Kinesis pipeline működik!")
```
