### 4.1 ETL script feltöltése S3-ba

```python
import boto3

REGION    = 'us-east-1'
NEPTUN    = "______"   # TODO
PREFIX    = f"bme-de-{NEPTUN.lower()}"
BUCKETS   = {
    "raw":       f"{PREFIX}-smart-city-raw",
    "processed": f"{PREFIX}-smart-city-processed",
}

sts        = boto3.client('sts', region_name=REGION)
ACCOUNT_ID = sts.get_caller_identity()["Account"]
LAB_ROLE   = f"arn:aws:iam::{ACCOUNT_ID}:role/LabRole"

s3         = boto3.client('s3',   region_name=REGION)
glue       = boto3.client('glue', region_name=REGION)

SCRIPT_KEY = "scripts/bronze_to_silver.py"
JOB_NAME   = f"{PREFIX}-sensor-bronze-to-silver"

with open("/home/jovyan/work/scripts/bronze_to_silver.py", "rb") as f:
    s3.put_object(
        Bucket=BUCKETS["raw"],
        Key=SCRIPT_KEY,
        Body=f.read(),
    )
print(f"✓ Script feltöltve: s3://{BUCKETS['raw']}/{SCRIPT_KEY}")
```

### 4.3 Job futtatása + státusz polling

```python
import time

response   = glue.start_job_run(JobName=JOB_NAME)
job_run_id = response["JobRunId"]
print(f"Job elindítva! Run ID: {job_run_id}")
print("Várakozás... (Glue Spark job indítása 2-4 percet vesz igénybe)")

while True:
    run    = glue.get_job_run(JobName=JOB_NAME, RunId=job_run_id)["JobRun"]
    status = run["JobRunState"]
    elapsed = run.get("ExecutionTime", 0)
    print(f"  [{elapsed:3d}s] Státusz: {status}")
    if status in ("SUCCEEDED", "FAILED", "ERROR", "STOPPED", "TIMEOUT"):
        break
    time.sleep(20)

if status != "SUCCEEDED":
    err = run.get("ErrorMessage", "ismeretlen hiba")
    raise RuntimeError(f"Job sikertelen ({status}): {err}")

print(f"\n✓ Job sikeresen lefutott! ({elapsed} másodperc)")
```

### 4.4 Parquet output ellenőrzése

```python
# Listázzuk a processed bucket tartalmát
paginator = s3.get_paginator('list_objects_v2')
pages     = paginator.paginate(Bucket=BUCKETS["processed"], Prefix="sensors/")

parquet_files = []
for page in pages:
    for obj in page.get('Contents', []):
        if obj['Key'].endswith('.parquet') or obj['Key'].endswith('.snappy.parquet'):
            parquet_files.append(obj)

print(f"Parquet fájlok ({len(parquet_files)} db):")
for pf in parquet_files[:10]:
    print(f"  {pf['Key']}  ({pf['Size']/1024:.1f} KB)")

assert len(parquet_files) > 0, "HIBA: Nem jöttek létre Parquet fájlok!"
print(f"\n✓ Parquet output sikeresen létrejött!")
```
