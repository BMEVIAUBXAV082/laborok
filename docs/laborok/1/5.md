    ```python
    import pyarrow.parquet as pq

    import time
    import matplotlib.pyplot as plt

    duck = duckdb.connect()

    RAW  = "/home/jovyan/data/extract/full"
    GOLD = "/home/jovyan/data/gold"

    # Staging és cleaned táblák újraépítése
    for t in ["customers", "products", "orders", "order_items"]:
        duck.execute(f"CREATE OR REPLACE TABLE stg_{t} AS SELECT * FROM read_parquet('{RAW}/{t}.parquet')")

    duck.execute("""
        CREATE OR REPLACE TABLE cleaned_orders AS
        SELECT o.*, c.name AS customer_name, c.segment AS customer_segment, c.city AS customer_city,
               CASE WHEN o.total_amount >= 100000 THEN 'nagy'
                    WHEN o.total_amount >= 20000 THEN 'közepes' ELSE 'kis' END AS order_size
        FROM stg_orders o JOIN stg_customers c ON o.customer_id = c.customer_id
        WHERE o.status != 'cancelled'
    """)
    print("Staging és cleaned táblák kész.")
    ```

### Gold réteg exportálása

3. Exportáljuk a transzformált táblákat Parquet formátumba:

    ```python
    gold_tables = [
        "cleaned_orders",
        "stg_customers",
        "stg_products",
        "stg_order_items",
    ]

    print("=== Gold réteg export ===\n")
    print(f"{'Tábla':<20} {'Sorok':>8} {'Méret (KB)':>12}")
    print("-" * 43)
    for table in gold_tables:
        out = f"{GOLD}/{table}.parquet"
        duck.execute(f"COPY {table} TO '{out}' (FORMAT PARQUET)")
        size_kb = os.path.getsize(out) / 1024
        cnt = duck.execute(f"SELECT COUNT(*) FROM {table}").fetchone()[0]
        print(f"  {table:<18} {cnt:>8,} {size_kb:>11.1f}")
    ```

### PostgreSQL vs DuckDB teljesítmény

4. Futtassuk ugyanazt az aggregációs lekérdezést mindkét motoron, és mérjük az időt:

    ```python
    query_pg = """
        SELECT p.category,
               COUNT(DISTINCT oi.order_id)           AS order_count,
               SUM(oi.quantity * oi.unit_price)::numeric(14,2) AS total_revenue
        FROM order_items oi
        JOIN products p ON oi.product_id = p.product_id
        GROUP BY p.category
        ORDER BY total_revenue DESC
    """

    # PostgreSQL
    start = time.time()
    pg_result = pd.read_sql(query_pg, engine)
    pg_ms = (time.time() - start) * 1000

    # DuckDB (gold Parquet)
    start = time.time()
    duck_result = duck.execute(f"""
        SELECT p.category,
               COUNT(DISTINCT oi.order_id)           AS order_count,
               SUM(oi.quantity * oi.unit_price)::NUMERIC(14,2) AS total_revenue
        FROM read_parquet('{GOLD}/stg_order_items.parquet') oi
        JOIN read_parquet('{GOLD}/stg_products.parquet') p ON oi.product_id = p.product_id
        GROUP BY p.category
        ORDER BY total_revenue DESC
    """).df()
    duck_ms = (time.time() - start) * 1000

    print("=== Teljesítmény összehasonlítás ===\n")
    print(f"  PostgreSQL:  {pg_ms:>8.1f} ms")
    print(f"  DuckDB:      {duck_ms:>8.1f} ms")
    print(f"  Arány:       {pg_ms/duck_ms:.1f}×")
    print()
    print("Eredmény (DuckDB):")
    print(duck_result.to_string(index=False))
    ```

    !!! note "Kis adathalmazok és teljesítmény"
        Kis adathalmazokon (ezres nagyságrendű sorok) a különbség mérsékelt lehet, mert a DuckDB indítási overhead-je relatíve nagy. Millió+ soros adatnál a DuckDB columnar, vektorizált végrehajtása jellemzően **10–100×-os** sebességelőnnyel rendelkezik aggregációs lekérdezéseknél.

    ```python
        SELECT
    ```

    ```python
    monthly_revenue = duck.execute(f"""
    ```

### Szegmens cohort elemzés

8. Vizsgáljuk meg, hogy az egyes ügyfélszegmensek mikor léptek be (első rendelés hónapja), és mekkora volt az összbevételük:

    ```python
    cohort = duck.execute(f"""
        WITH first_order AS (
            SELECT
                customer_id,
                DATE_TRUNC('month', MIN(order_date))::DATE AS cohort_month
            FROM read_parquet('{GOLD}/cleaned_orders.parquet')
            GROUP BY customer_id
        ),
        cohort_revenue AS (
            SELECT
                fo.cohort_month,
                c.segment                             AS customer_segment,
                COUNT(DISTINCT o.customer_id)         AS customers,
                SUM(o.total_amount)::NUMERIC(14,2)    AS total_revenue
            FROM read_parquet('{GOLD}/cleaned_orders.parquet') o
            JOIN first_order fo ON o.customer_id = fo.customer_id
            JOIN read_parquet('{GOLD}/stg_customers.parquet') c
              ON o.customer_id = c.customer_id
            GROUP BY fo.cohort_month, c.segment
        )
        SELECT *,
               ROUND(total_revenue / customers, 0)::NUMERIC(14,0) AS revenue_per_customer
        FROM cohort_revenue
        ORDER BY cohort_month DESC, customer_segment
        LIMIT 20
    """).df()

    print("=== Szegmens cohort elemzés (utolsó cohort-ok) ===\n")
    print(cohort.to_string(index=False))
    ```

    !!! tip "Gold réteg értéke"
        A gold réteg táblái analitikára készen állnak: tisztítva, joinolva, aggregálva. Ezekből közvetlenül táplálhatók BI eszközök (Tableau, Power BI, Superset) vagy adattárházak (BigQuery, Snowflake). A Parquet formátum biztosítja, hogy bármely analitikai motor (DuckDB, Spark, Athena) hatékonyan el tudja olvasni.
