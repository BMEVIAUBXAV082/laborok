### 2.3 Hive-partícionált feltöltés S3-ba

```python
import io
import boto3

s3 = boto3.client('s3', region_name='us-east-1')

for sensor_type in df['device_type'].unique():
    subset = df[df['device_type'] == sensor_type].copy()

    buf = io.StringIO()
    subset.to_csv(buf, index=False)
    csv_bytes = buf.getvalue().encode('utf-8')

    # Hive partition path: sensor_type=X/year=2025/month=01/
    s3_key = f"sensors/sensor_type={sensor_type}/year=2025/month=01/data.csv"

    s3.put_object(
        Bucket=BUCKETS["raw"],    # a raw zóna neve
        Key=s3_key,               # TODO: ellenőrizd, hogy az s3_key változót használod
        Body=csv_bytes,
        ContentType='text/csv',
    )
    print(f"✓ Feltöltve: s3://{BUCKETS['raw']}/{s3_key}  ({len(subset):,} sor)")
```

### 2.4 Ellenőrzés

```python
# Listázzuk az S3-ba kerülő objektumokat
paginator = s3.get_paginator('list_objects_v2')
pages = paginator.paginate(Bucket=BUCKETS["raw"], Prefix="sensors/")

total_size = 0
for page in pages:
    for obj in page.get('Contents', []):
        size_kb = obj['Size'] / 1024
        total_size += obj['Size']
        print(f"  {obj['Key']}  ({size_kb:.1f} KB)")

print(f"\nÖsszes méret: {total_size/1024:.1f} KB")
assert total_size > 0, "HIBA: Nem töltődött fel semmi!"
print("✓ Feltöltés sikeres!")
```
