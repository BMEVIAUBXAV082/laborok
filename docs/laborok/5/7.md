### 7.1 Lambda function kódjának betöltése + ZIP csomagolás

```python
import boto3, io, zipfile

REGION      = 'us-east-1'
NEPTUN      = "______"   # TODO
PREFIX      = f"bme-de-{NEPTUN.lower()}"
STREAM_NAME = f"{PREFIX}-smart-city-events"
BUCKETS     = {"streaming": f"{PREFIX}-smart-city-streaming"}

sts        = boto3.client('sts',      region_name=REGION)
lmb        = boto3.client('lambda',   region_name=REGION)
kinesis    = boto3.client('kinesis',  region_name=REGION)
firehose   = boto3.client('firehose', region_name=REGION)

ACCOUNT_ID = sts.get_caller_identity()["Account"]
LAB_ROLE   = f"arn:aws:iam::{ACCOUNT_ID}:role/LabRole"

FN_NAME = f"{PREFIX}-iot-processor"

# Lambda kód betöltése a scripts/ mappából
with open("/home/jovyan/work/scripts/lambda_processor.py", "r") as f:
    lambda_src = f.read()

# ZIP készítés memóriában
buf = io.BytesIO()
with zipfile.ZipFile(buf, "w", zipfile.ZIP_DEFLATED) as zf:
    zf.writestr("lambda_function.py", lambda_src)
zip_bytes = buf.getvalue()

print(f"Lambda ZIP mérete: {len(zip_bytes)} byte")
print("Lambda kód (első 10 sor):")
print("\n".join(lambda_src.split("\n")[:10]))
```

### 7.2 Lambda függvény létrehozása

```python
import time

try:
    resp = lmb.create_function(
        FunctionName=None,         # TODO: FN_NAME
        Runtime="python3.12",
        Role=None,                 # TODO: LAB_ROLE
        Handler="lambda_function.lambda_handler",
        Code={"ZipFile": zip_bytes},
        Timeout=60,
        MemorySize=128,
        Description="Smart City IoT esemény validátor és enricher",
    )
    fn_arn = resp["FunctionArn"]
    print(f"✓ Lambda létrehozva: {FN_NAME}")
    print(f"  ARN: {fn_arn}")
except lmb.exceptions.ResourceConflictException:
    info   = lmb.get_function(FunctionName=FN_NAME)
    fn_arn = info["Configuration"]["FunctionArn"]
    print(f"  Lambda már létezik: {FN_NAME}")

# Megvárjuk, hogy Active állapotba kerüljön
for _ in range(20):
    info   = lmb.get_function_configuration(FunctionName=FN_NAME)
    state  = info.get("State", "Unknown")
    if state == "Active":
        break
    time.sleep(3)
print(f"  Státusz: {state}")
```

### 7.4 Kinesis Firehose delivery stream létrehozása

```python
FH_NAME = f"{PREFIX}-iot-firehose"

# Streaming bucket ARN-ja
streaming_arn = f"arn:aws:s3:::{BUCKETS['streaming']}"

try:
    firehose.create_delivery_stream(

            "BucketARN": streaming_arn,
            "Prefix":    "sensors/year=!{timestamp:yyyy}/month=!{timestamp:MM}/day=!{timestamp:dd}/",
            "ErrorOutputPrefix": "errors/",
            "BufferingHints": {
                "SizeInMBs":         None,   # TODO: 5  (MB)
```

### 7.5 Producer futtatása + S3 ellenőrzése

```python
import subprocess, sys

print("Streaming producer indítása (100 esemény, 0.3s delay)...")
print("Ez kb. 30 másodpercet vesz igénybe...\n")

result = subprocess.run(
    [sys.executable,
     "/home/jovyan/work/scripts/streaming_producer.py",
     "--stream", STREAM_NAME,
     "--count",  "100",
     "--delay",  "0.3",
     "--region", REGION],
    capture_output=True, text=True,
)
print(result.stdout)
if result.returncode != 0:
    print("STDERR:", result.stderr)

print("\nVárunk 70 másodpercet a Firehose buffer teljesítéséig...")
time.sleep(70)

# S3 streaming bucket ellenőrzése
paginator = boto3.client('s3', region_name=REGION).get_paginator('list_objects_v2')
pages = paginator.paginate(Bucket=BUCKETS["streaming"], Prefix="sensors/")

files = []
for page in pages:
    files.extend(page.get('Contents', []))

print(f"\nS3 streaming mappa ({len(files)} fájl):")
for f in files:
    print(f"  {f['Key']}  ({f['Size']} byte)")

if len(files) == 0:
    print("  (A Firehose 60 másodpercenként tölt fel – várj még egy kicsit és futtasd újra az ellenőrzést)")
```
